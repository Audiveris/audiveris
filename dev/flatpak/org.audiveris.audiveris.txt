app-id: org.audiveris.audiveris

  append-path: "/usr/lib/sdk/${OPEN_JDK}/bin"

  - --share=network
  # X11 + XShm access

  # Needs to save files locally

    build-commands:
      # Gradle wrapper
      - ln -f ${GRADLE_ZIP} gradle/wrapper/${GRADLE_ZIP}
      # NOTA: mind the fact that expand() will modify the backslashes, so double them!

      # Copy desktop and metainfo
      - install -m 0644 -D org.audiveris.audiveris.desktop /app/share/applications/org.audiveris.audiveris.desktop

      - sed -i -f add-tessdata-prefix.sed /app/bin/Audiveris
